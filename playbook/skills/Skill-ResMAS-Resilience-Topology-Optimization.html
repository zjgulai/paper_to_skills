"""
ResMASйҹ§жҖ§жӢ“жү‘дјҳеҢ–зі»з»ҹ
еҠҹиғҪпјҡGNNйҹ§жҖ§йў„жөӢ(з®ҖеҢ–зүҲ) + жӢ“жү‘з”ҹжҲҗ + жӢ“жү‘ж„ҹзҹҘPrompt + йҹ§жҖ§иҜ„дј°
еҹәдәҺ arXiv:2601.04694 (2026)
"""
import numpy as np
from dataclasses import dataclass, field
from typing import List, Dict, Optional, Tuple, Set
from enum import Enum
import warnings
warnings.filterwarnings('ignore')


class TopologyType(Enum):
    LINEAR = "linear"       # AвҶ’BвҶ’CпјҲжңҖи„Ҷејұпјү
    STAR = "star"           # AвҶ’(B,C,D)пјҲдёӯзӯүпјү
    HIERARCHICAL = "hierarchical"  # AвҶ’(BвҶ”C)вҶ’DпјҲжңҖејәйҹ§пјү
    FLAT = "flat"           # AвҶ”BвҶ”CпјҲзҺҜеҪўпјҢдёӯзӯүпјү
    CUSTOM = "custom"


@dataclass
class AgentNode:
    """MASжӢ“жү‘дёӯзҡ„AgentиҠӮзӮ№"""
    agent_id: str
    role: str
    failure_rate: float = 0.0      # еҪ“еүҚеӨұж•ҲзҺҮ
    base_quality: float = 0.85     # ж— еӨұж•Ҳж—¶зҡ„еҹәзЎҖиҙЁйҮҸ


@dataclass
class MASTopology:
    """MASжӢ“жү‘з»“жһ„"""
    topology_type: TopologyType
    agents: List[AgentNode]
    edges: List[Tuple[str, str]]   # (from_agent, to_agent) жңүеҗ‘иҫ№

    def get_neighbors(self, agent_id: str) -> List[str]:
        """иҺ·еҸ–Agentзҡ„зӣҙжҺҘдёҠжёёйӮ»еұ…"""
        return [src for src, dst in self.edges if dst == agent_id]

    def get_hub_agents(self) -> List[str]:
        """иҜҶеҲ«Hub AgentпјҲиў«еӨҡдёӘAgentдҫқиө–пјү"""
        in_degree = {a.agent_id: 0 for a in self.agents}
        for _, dst in self.edges:
            in_degree[dst] += 1
        return [aid for aid, deg in in_degree.items() if deg >= 2]

    @property
    def adjacency_features(self) -> np.ndarray:
        """жҸҗеҸ–жӢ“жү‘зү№еҫҒеҗ‘йҮҸпјҲGNNиҫ“е…Ҙзҡ„з®ҖеҢ–зүҲпјү"""
        n = len(self.agents)
        # зү№еҫҒпјҡ[е№іеқҮеәҰ, жңҖеӨ§е…ҘеәҰ, иҒҡзұ»зі»ж•°иҝ‘дјј, еұӮж¬Ўж·ұеәҰ, жҳҜеҗҰжңүеҸҢеҗ‘иҫ№]
        out_degrees = {a.agent_id: 0 for a in self.agents}
        in_degrees = {a.agent_id: 0 for a in self.agents}
        bidirectional = set()

        for src, dst in self.edges:
            out_degrees[src] += 1
            in_degrees[dst] += 1
