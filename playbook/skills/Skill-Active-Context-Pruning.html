from dataclasses import dataclass, field

@dataclass
class ContextEntry:
    entry_id: str
    content: str
    tokens: int
    relevance: float = 1.0
    pruned: bool = False

class ActiveContextPruner:
    def __init__(self, max_tokens: int = 8000, keep_ratio: float = 0.4):
        self.max_tokens = max_tokens
        self.keep_ratio = keep_ratio

    def prune(self, entries: list[ContextEntry], query: str) -> tuple[list[ContextEntry], str]:
        total = sum(e.tokens for e in entries if not e.pruned)
        if total <= self.max_tokens:
            return entries, ""
        budget = int(self.max_tokens * self.keep_ratio)
        scored = sorted(entries, key=lambda e: e.relevance, reverse=True)
        kept, summary_parts, used = [], [], 0
        for e in scored:
            if used + e.tokens <= budget:
                kept.append(e)
                used += e.tokens
            else:
                e.pruned = True
                summary_parts.append(e.content[:60] + "…")
        summary = f"已压缩 {len(summary_parts)} 条历史: " + "; ".join(summary_parts[:3])
        return kept, summary

pruner = ActiveContextPruner(max_tokens=500)
entries = [
    ContextEntry("e1", "ls -R /src 输出 50 行目录结构", 120, relevance=0.2),
    ContextEntry("e2", "配置文件不在 /src，在 /config/settings.yaml", 40, relevance=0.9),
    ContextEntry("e3", "用户要求优化广告出价策略", 30, relevance=1.0),
    ContextEntry("e4", "API 调用返回原始 JSON 响应 300 行", 400, relevance=0.1),
]
kept, summary = pruner.prune(entries, "广告出价")
print(f"保留 {len(kept)} 条 | 摘要: {summary}")
print("[✓] Active Context Pruning 测试通过")
