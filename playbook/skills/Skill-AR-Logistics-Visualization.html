import numpy as np
from dataclasses import dataclass, field
from typing import Dict, List, Optional, Tuple
from datetime import datetime, timedelta
from enum import Enum
import json


class PackageStatus(Enum):
    PICKED_UP = "picked_up"
    IN_TRANSIT = "in_transit"
    CUSTOMS_PENDING = "customs_pending"
    CUSTOMS_CLEARED = "customs_cleared"
    OUT_FOR_DELIVERY = "out_for_delivery"
    DELIVERED = "delivered"
    EXCEPTION = "exception"


class NodeType(Enum):
    ORIGIN_HUB = "origin_hub"
    DEPARTURE_AIRPORT = "departure_airport"
    ARRIVAL_AIRPORT = "arrival_airport"
    CUSTOMS = "customs"
    DOMESTIC_HUB = "domestic_hub"
    LAST_MILE = "last_mile"
    DELIVERED = "delivered"


@dataclass
class LogisticsEvent:
    """物流事件节点"""
    timestamp: datetime
    raw_text: str
    location: Tuple[float, float]      # (lat, lon)
    node_type: NodeType
    status: PackageStatus
    customs_cleared: Optional[bool] = None
    carrier: str = "unknown"
    metadata: Dict = field(default_factory=dict)


@dataclass
class ARTrackPoint:
    """AR 可渲染轨迹点"""
    lat: float
    lon: float
    altitude: float = 10000.0          # 飞行段高度（米）
    color: str = "green"               # green/yellow/red
    anomaly_score: float = 0.0
    eta_distribution: Tuple[float, float] = (1.0, 1.0)   # Beta(α, β)
    label: str = ""
